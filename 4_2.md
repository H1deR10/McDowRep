# ER-диаграмма: Социальная сеть

```mermaid
erDiagram
    %% --- Сущности ---
    USERS {
        int id PK
        string name
        string email
    }

    POSTS {
        int id PK
        string content
        int authorId FK
    }

    COMMENTS {
        int id PK
        string content
        int postId FK
        int authorId FK
    }

    LIKES {
        int userId FK
        int postId FK
    }

    SUBSCRIPTIONS {
        int followerId FK
        int followingId FK
    }

    %% --- Связи ---
    USERS ||--o{ POSTS : "создает"
    USERS ||--o{ COMMENTS : "пишет"
    POSTS ||--o{ COMMENTS : "содержит"
    USERS ||--o{ LIKES : "ставит"
    POSTS ||--o{ LIKES : "получает"
    USERS ||--o{ SUBSCRIPTIONS : "подписан на"
    USERS ||--o{ SUBSCRIPTIONS : "имеет подписчиков"
